create procedure company_insert
@Pcompany_id  int,
@Pcompany_name  varchar(20),
@Pcompany_telephone  varchar(20),
@Pcompany_mail  varchar(50),
@Pcompany_address  varchar(300),
@Pcompany_introduction  varchar(500),
@Pcompany_password  varchar(20),
as
begin
INSERT INTO company (company_id,company_name,company_telephone,company_mail,company_address,company_introduction,company_password) VALUES (@Pcompany_id,@Pcompany_name,@Pcompany_telephone,@Pcompany_mail,@Pcompany_address,@Pcompany_introduction,@Pcompany_password)
end
create procedure bid_insert
@Pbid_id  int,
@Ppurchase_id  int,

@Pbid_title  varchar(50),
@Pbid_device_name  varchar(50),
@Pbid_time  datetime,
@Pbid_number  int,
@Pbid_introduction  varchar(1500),
@Pbid_content  text,
as
begin
INSERT INTO bid (bid_id,purchase_id,vendor_id,bid_title,bid_device_name,bid_time,bid_number,bid_introduction,bid_content) VALUES (@Pbid_id,@Ppurchase_id,@Pvendor_id,@Pbid_title,@Pbid_device_name,@Pbid_time,@Pbid_number,@Pbid_introduction,@Pbid_content)
end
